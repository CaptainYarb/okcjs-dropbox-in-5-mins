Installation 1/2

Get a Drop-ins app key from Dropbox

<script type="text/javascript" src="https://www.dropbox.com/static/api/2/dropins.js"
	id="dropboxjs" data-app-key="YOUR_APP_KEY"></script>

Installation 2/2

var chooser = Dropbox.createChooseButton({
	success: function(files) {
		console.log(files); // will return an array of files
	},
	cancel: function(){ ... },
	linkType: "direct", // 'preview' for a sharable link
	multiselect: false,
	extensions: [".pdf", ".png", ".jpg"]
});
document.getElementById("someDiv").appendChild(chooser);

Installation 2/2

var saver = Dropbox.createSaveButton({
	success: function() { ... },
	error: function(err) { ... },
	progress: function(prog) {
		console.log('Saving Progress: [%s]', Math.ceil(prog * 100));
	},
	files: [
		{url: "http://example.com/sales-report.pdf", filename: "Sales Report.pdf"}
	]
});
document.getElementById("someDiv").appendChild(saver);
